386 | function buildSubgraph(nodeId, allNodes, allEdges, depth) {
387 |     const visited = new Set([nodeId]);
388 |     const dependents = [];
389 |     const dependencies = [];
390 |     const queue = [[nodeId, 0]];
391 |     
392 |     while (queue.length > 0) {
393 |         const [current, d] = queue.shift();
394 |         if (d >= depth) continue;
395 |         
396 |         // Find dependents (nodes that depend on current)
397 |         for (const edge of allEdges) {
398 |             if (edge.target === current && !visited.has(edge.source)) {
399 |                 visited.add(edge.source);
400 |                 const node = allNodes.find(n => n.id === edge.source);
401 |                 if (node) {
402 |                     dependents.push(node);
403 |                     queue.push([edge.source, d + 1]);
404 |                 }
405 |             }
406 |         }
407 |         
408 |         // Find dependencies (nodes that current depends on)
409 |         for (const edge of allEdges) {
410 |             if (edge.source === current && !visited.has(edge.target)) {
411 |                 visited.add(edge.target);
412 |                 const node = allNodes.find(n => n.id === edge.target);
413 |                 if (node) {
414 |                     dependencies.push(node);
415 |                     queue.push([edge.target, d + 1]);
416 |                 }
417 |             }
418 |         }
419 |     }
420 |     
421 |     // Filter edges to only include those between visible nodes
422 |     const visibleEdges = allEdges.filter(e =>
423 |         visited.has(e.source) && visited.has(e.target)
424 |     );
425 |     
426 |     return {
427 |         selected_node: allNodes.find(n => n.id === nodeId),
428 |         dependents,
429 |         dependencies,
430 |         edges: visibleEdges,
431 |         depth
432 |     };
